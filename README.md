│
├── config/                    # File konfigurasi proyek

│
├── docs/                      # Dokumentasi tambahan proyek

├── .gitignore                 # File untuk menentukan file/folder yang diabaikan oleh Git
│
├── LICENSE                    # Lisensi proyek
│
└── README.md                  # Dokumentasi proyek
